Problem Statement
- Job Portal  (Only for Company & Job)

Design and implement a Job Portal Management Service using Spring Boot, Spring Data JPA, and REST APIs, where companies can post jobs , change the job status etc.

Company (Represents an organization that can post job openings)

Attributes 
 - id (Long - PK) - auto increment
 - name (unique)
 - email (unique)
 - location
 - industry (enum - Education , Pharmaceutical , Hospitality , Information Technology (IT)  Banking , Healthcare)
 - created on :  date
 
Job (Represents a job opening posted by a company)
Attributes
-	id (Long, PK)  - auto increment
-	title 
-	description
-	salary
-	location
-	jobType (enum FULL_TIME / PART_TIME / CONTRACT)
-	postedDate
-        status (enum - AVAILABLE | UNAVAILABLE)
-	company (Many-to-One)


Add sample data for some  companies (directly in DB)

Objectives
APIs 
1.  List all companies by specified industry type & location

2.  Post new job opening
Payload
 company id , title , description ,salary,location, jobType

3. List all job openings posted by companies , from the specified location
Input  - location

4. Delete all jobs of the specified job type , of specified company
 Input - job type , company name

5. Update salaries of the job , posted by a specific company , for specific job title
Input - company id , job title

6. List all the jobs , from specific industry type & having minimum specified salary.
Input - industry type , min salary

7. Comany needs to update a job status to unavailable (position filled)
Input - company name & job title

-	Use MySQL Relational Database to save resource.
- None of the fields in jobs table are Nullable.
-	Rest URL should be named as per REST best practices.
-	Response codes should be as per REST guidelines (201 for resource creation, 404 for not found, 200 for ok...)
-	Error handling in case of failures. (Optional : Global Exception Handling)
Note:- You have to test all API's on Postman or Swagger

Job Portal Management Service – Spring Boot + JPA + MySQL

package com.example.jobportal.entity;

import jakarta.persistence.*;
import java.time.LocalDate;
import java.util.List;

@Entity
@Table(name = "companies")
public class Company {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(unique = true, nullable = false)
    private String name;

    @Column(unique = true, nullable = false)
    private String email;

    @Column(nullable = false)
    private String location;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private IndustryType industry;

    private LocalDate createdOn;

    @OneToMany(mappedBy = "company", cascade = CascadeType.ALL)
    private List<Job> jobs;

    @PrePersist
    public void prePersist() {
        createdOn = LocalDate.now();
    }

    // Getters and Setters
}
Job.java
package com.example.jobportal.entity;

import jakarta.persistence.*;
import java.time.LocalDate;

@Entity
@Table(name = "jobs")
public class Job {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(nullable = false)
    private String title;

    @Column(nullable = false)
    private String description;

    @Column(nullable = false)
    private Double salary;

    @Column(nullable = false)
    private String location;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private JobType jobType;

    private LocalDate postedDate;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private JobStatus status;

    @ManyToOne
    @JoinColumn(name = "company_id", nullable = false)
    private Company company;

    @PrePersist
    public void prePersist() {
        postedDate = LocalDate.now();
        status = JobStatus.AVAILABLE;
    }

    // Getters and Setters
}

JobService.java
package com.example.jobportal.service;

import com.example.jobportal.dto.JobRequestDto;
import com.example.jobportal.dto.SalaryUpdateDto;
import com.example.jobportal.entity.*;
import com.example.jobportal.exception.ResourceNotFoundException;
import com.example.jobportal.repository.CompanyRepository;
import com.example.jobportal.repository.JobRepository;
import org.springframework.stereotype.Service;

import java.util.List;

@Service
public class JobService {

    private final JobRepository jobRepository;
    private final CompanyRepository companyRepository;

    public JobService(JobRepository jobRepository,
                      CompanyRepository companyRepository) {

        this.jobRepository = jobRepository;
        this.companyRepository = companyRepository;
    }

    // Post Job
    public Job createJob(JobRequestDto dto) {

        Company company = companyRepository.findById(dto.getCompanyId())
                .orElseThrow(() ->
                        new ResourceNotFoundException("Company not found"));

        Job job = new Job();

        job.setTitle(dto.getTitle());
        job.setDescription(dto.getDescription());
        job.setSalary(dto.getSalary());
        job.setLocation(dto.getLocation());
        job.setJobType(dto.getJobType());
        job.setCompany(company);

        return jobRepository.save(job);
    }

    // Get Jobs by Location
    public List<Job> getJobsByLocation(String location) {
        return jobRepository.findByLocation(location);
    }

    // Delete Jobs
    public void deleteJobs(JobType jobType,
                           String companyName) {

        jobRepository.deleteByJobTypeAndCompany_Name(
                jobType,
                companyName
        );
    }

    // Update Salary
    public Job updateSalary(SalaryUpdateDto dto) {

        Job job = jobRepository.findByCompany_IdAndTitle(
                dto.getCompanyId(),
                dto.getJobTitle()
        );

        if (job == null) {
            throw new ResourceNotFoundException("Job not found");
        }

        job.setSalary(dto.getSalary());

        return jobRepository.save(job);
    }

    // Get jobs by industry & min salary
    public List<Job> getJobsByIndustryAndSalary(
            IndustryType industry,
            Double minSalary) {

        return jobRepository
                .findByCompany_IndustryAndSalaryGreaterThanEqual(
                        industry,
                        minSalary
                );
    }

    // Update status
    public Job updateJobStatus(String companyName,
                               String title) {

        Job job = jobRepository.findByCompany_NameAndTitle(
                companyName,
                title
        );

        if (job == null) {
            throw new ResourceNotFoundException("Job not found");
        }

        job.setStatus(JobStatus.UNAVAILABLE);

        return jobRepository.save(job);
    }
}

JobController.java
package com.example.jobportal.controller;

import com.example.jobportal.dto.JobRequestDto;
import com.example.jobportal.dto.SalaryUpdateDto;
import com.example.jobportal.entity.*;
import com.example.jobportal.service.JobService;
import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/api/jobs")
public class JobController {

    private final JobService jobService;

    public JobController(JobService jobService) {
        this.jobService = jobService;
    }

    // 2. Post New Job
    @PostMapping
    @ResponseStatus(HttpStatus.CREATED)
    public Job createJob(@RequestBody JobRequestDto dto) {
        return jobService.createJob(dto);
    }

    // 3. List Jobs by Location
    @GetMapping("/location/{location}")
    public List<Job> getJobsByLocation(
            @PathVariable String location) {

        return jobService.getJobsByLocation(location);
    }

    // 4. Delete Jobs
    @DeleteMapping
    public String deleteJobs(
            @RequestParam JobType jobType,
            @RequestParam String companyName) {

        jobService.deleteJobs(jobType, companyName);

        return "Jobs deleted successfully";
    }

    // 5. Update Salary
    @PutMapping("/salary")
    public Job updateSalary(
            @RequestBody SalaryUpdateDto dto) {

        return jobService.updateSalary(dto);
    }

    // 6. Jobs by Industry & Salary
    @GetMapping("/filter")
    public List<Job> getJobs(
            @RequestParam IndustryType industry,
            @RequestParam Double minSalary) {

        return jobService.getJobsByIndustryAndSalary(
                industry,
                minSalary
        );
    }

    // 7. Update Job Status
    @PutMapping("/status")
    public Job updateStatus(
            @RequestParam String companyName,
            @RequestParam String jobTitle) {

        return jobService.updateJobStatus(
                companyName,
                jobTitle
        );
    }
}

GlobalExceptionHandler.java
package com.example.jobportal.exception;

import org.springframework.http.HttpStatus;
import org.springframework.web.bind.annotation.*;

import java.util.Map;

@RestControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(ResourceNotFoundException.class)
    @ResponseStatus(HttpStatus.NOT_FOUND)

            ResourceNotFoundException ex) {
